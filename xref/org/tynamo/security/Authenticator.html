7   /**
8    * Replacement for Shiro's Authenticator interface while waiting for https://issues.apache.org/jira/browse/SHIRO-233 to be
9    * satisfactorily resolved
10   *
11   */
12  public interface Authenticator extends org.apache.shiro.authc.Authenticator, AuthenticationListenerRegistrar {
13    /**
14     * Sets all realms used by this Authenticator, providing PAM (Pluggable Authentication Module) configuration. 
15     * 
16     * The operation is copied from org.apache.shiro.authc.pam.ModularRealmAuthenticator. Shiro's design is less than ideal for proxied 
17     * interface-based systems because it internally relies on downcasting to specific types. Not all Authenticators need realms but 
18     * ModularRealmAuthenticator has its own reference to the realms collection (separate from securityManager) so make the setter 
19     * available here.
20     *
21     * @param realms the realms to consult during authentication attempts.
22     */
23    public void setRealms(Collection<Realm> realms);
24  }
