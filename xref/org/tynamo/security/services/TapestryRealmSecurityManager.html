12  import org.tynamo.security.Authenticator;
13  
14  import java.util.Collection;
15  
16  /**
17   * This class is needed to point out the right constructor to use (from the three available in
18   * DefaultWebSecurityManager) for tapestry-ioc and to allow injecting dependencies
19   */
20  public class TapestryRealmSecurityManager extends DefaultWebSecurityManager {
21  
22  	// Could easily make sessionStorageevaluator and sessionManager provided services as well, add as needed
23  	public TapestryRealmSecurityManager(Authenticator authenticator, SubjectFactory subjectFactory, RememberMeManager rememberMeManager, final Collection<Realm> realms) {
24  		super();
25  		authenticator.setRealms(realms);
26  		setAuthenticator(authenticator);
27  		((DefaultSubjectDAO) this.subjectDAO).setSessionStorageEvaluator(new DefaultWebSessionStorageEvaluator());
28  		setSubjectFactory(subjectFactory);
29  		setRememberMeManager(rememberMeManager);
30  		setSessionManager(new ServletContainerSessionManager());
31  		setRealms(realms);
32  	}
33  
34  	@PostInjection
35  	public void listenForShutdown(RegistryShutdownHub hub) {
36  		hub.addRegistryShutdownListener(new Runnable() {
37  			public void run() {
38  				destroy();
39  			}
40  		});
41  	}
42  
43  }
