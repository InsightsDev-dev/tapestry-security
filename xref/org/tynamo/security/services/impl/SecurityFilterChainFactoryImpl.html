26  // Eager load since this service initializes the global filter defaults
27  @EagerLoad
28  public class SecurityFilterChainFactoryImpl implements SecurityFilterChainFactory {
29  
30  	private final Logger logger;
31  
32  	private final PipelineBuilder builder;
33  
34  	private final LoginContextService loginContextService;
35  
36  	private final AntPathMatcher antPathMatcher = new AntPathMatcher();
37  	private final RegExPatternMatcher regExPatternMatcher= new RegExPatternMatcher();
38  
39  	public SecurityFilterChainFactoryImpl(Logger logger, PipelineBuilder builder, LoginContextService loginContextService,
40        @Inject @Symbol(SymbolConstants.TAPESTRY_VERSION) String tapestryVersion,
41        @Inject @Symbol(SecuritySymbols.SUCCESS_URL) String successUrl,
42        @Inject @Symbol(SecuritySymbols.LOGIN_URL) String loginUrl,
43        @Inject @Symbol(SecuritySymbols.UNAUTHORIZED_URL) String unauthorizedUrl
44  	) {
45  		this.builder = builder;
46  		this.logger = logger;
47  		this.loginContextService = loginContextService;
48  		AccessControlFilter.TAPESTRY_VERSION = tapestryVersion;
49  		AccessControlFilter.LOGIN_URL = loginUrl;
50  		AccessControlFilter.SUCCESS_URL = successUrl;
51  		AccessControlFilter.UNAUTHORIZED_URL = unauthorizedUrl;
52  	}
53  
54  	public SecurityFilterChain.Builder createChain(String path) {
55  		return new SecurityFilterChain.Builder(logger, builder, path, antPathMatcher);
56  	}
57  
58  	public SecurityFilterChain.Builder createChain(String pattern, PatternMatcher patternMatcher) {
59  		return new SecurityFilterChain.Builder(logger, builder, pattern, patternMatcher);
60  	}
61  
62  	public SecurityFilterChain.Builder createChainWithAntPath(String path) {
63  		return new SecurityFilterChain.Builder(logger, builder, path, antPathMatcher);
64  	}
65  
66  	public SecurityFilterChain.Builder createChainWithRegEx(String pattern) {
67  		return new SecurityFilterChain.Builder(logger, builder, pattern, regExPatternMatcher);
68  	}
