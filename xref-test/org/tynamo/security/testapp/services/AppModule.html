71  	public static void contributeApplicationDefaults(MappedConfiguration<String, String> configuration) {
72  		// Contributions to ApplicationDefaults will override any contributions to
73  		// FactoryDefaults (with the same key). Here we're restricting the supported
74  		// locales to just "en" (English). As you add localised message catalogs and other assets,
75  		// you can extend this list of locales (it's a comma separated series of locale names;
76  		// the first locale name is the default when there's no reasonable match).
77  
78  		configuration.add(SymbolConstants.SUPPORTED_LOCALES, "en, fi_FI");
79  
80  
81  		// The factory default is true but during the early stages of an application
82  		// overriding to false is a good idea. In addition, this is often overridden
83  		// on the command line as -Dtapestry.production-mode=false
84  		configuration.add(SymbolConstants.PRODUCTION_MODE, "false");
85  
86  		// The application version number is incorprated into URLs for some
87  		// assets. Web browsers will cache assets because of the far future expires
88  		// header. If existing assets are changed, the version number should also
89  		// change, to force the browser to download new versions.
90  		configuration.add(SymbolConstants.APPLICATION_VERSION, "0.0.1-SNAPSHOT");
91  
92  		configuration.add(SecuritySymbols.SUCCESS_URL, "/index");
93  
94  		// HTMLUnit gets confused with prototype's readAttribute:
95  		// net.sourceforge.htmlunit.corejs.javascript.EcmaError: TypeError: Cannot find function readAttribute in object [object
96  		// HTMLBodyElement]. (http://localhost:8180/test/modules.gz/t5/core/dom.js#137)
97  		// and anyway, jquery is by far the more popular one nowadays
98  		configuration.add(SymbolConstants.JAVASCRIPT_INFRASTRUCTURE_PROVIDER, "jquery");
99  	}

102 	/**
103 	 * This is a service definition, the service will be named "TimingFilter". The interface,
104 	 * RequestFilter, is used within the RequestHandler service pipeline, which is built from the
105 	 * RequestHandler service configuration. Tapestry IoC is responsible for passing in an
106 	 * appropriate Logger instance. Requests for static resources are handled at a higher level, so
107 	 * this filter will only be invoked for Tapestry related requests.
108 	 * <p/>
109 	 * <p/>
110 	 * Service builder methods are useful when the implementation is inline as an inner class
111 	 * (as here) or require some other kind of special initialization. In most cases,
112 	 * use the static bind() method instead.
113 	 * <p/>
114 	 * <p/>
115 	 * If this method was named "build", then the service id would be taken from the
116 	 * service interface and would be "RequestFilter".  Since Tapestry already defines
117 	 * a service named "RequestFilter" we use an explicit service id that we can reference
118 	 * inside the contribution method.
119 	 */
120 	public RequestFilter buildTimingFilter(final Logger log) {
121 		return new RequestFilter() {
122 			@Override
123 			public boolean service(Request request, Response response, RequestHandler handler)
124 					throws IOException {
125 				long startTime = System.currentTimeMillis();
126 
127 				try {
128 					// The responsibility of a filter is to invoke the corresponding method
129 					// in the handler. When you chain multiple filters together, each filter
130 					// received a handler that is a bridge to the next filter.
131 
132 					return handler.service(request, response);
133 				}
134 				finally {
135 					long elapsed = System.currentTimeMillis() - startTime;
136 
137 					log.info(String.format("Request time: %d ms", elapsed));
138 				}
139 			}
140 		};
141 	}

143 	/**
144 	 * This is a contribution to the RequestHandler service configuration. This is how we extend
145 	 * Tapestry using the timing filter. A common use for this kind of filter is transaction
146 	 * management or security. The @Local annotation selects the desired service by type, but only
147 	 * from the same module.  Without @Local, there would be an error due to the other service(s)
148 	 * that implement RequestFilter (defined in other modules).
149 	 */
150 	public void contributeRequestHandler(OrderedConfiguration<RequestFilter> configuration,
151 	                                     @Local RequestFilter filter) {
152 		// Each contribution to an ordered configuration has a name, When necessary, you may
153 		// set constraints to precisely control the invocation order of the contributed filter
154 		// within the pipeline.
155 
156 		configuration.add("Timing", filter);
157 	}
