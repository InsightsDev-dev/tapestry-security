068        public static void bind(final ServiceBinder binder)
069        {
070
071                binder.bind(WebSecurityManager.class, TapestryRealmSecurityManager.class);
072                // TYNAMO-155 It's not enough to identify ModularRealmAuthenticator by it's Authenticator interface only
073                // because Shiro tests if the object is an instanceof LogoutAware to call logout handlers
074                binder.bind(ModularRealmAuthenticator.class);
075                binder.bind(SubjectFactory.class, DefaultWebSubjectFactory.class);
076                binder.bind(HttpServletRequestFilter.class, SecurityConfiguration.class).withId("SecurityConfiguration");
077                binder.bind(ClassInterceptorsCache.class, ClassInterceptorsCacheImpl.class);
078                binder.bind(SecurityService.class, SecurityServiceImpl.class);
079                binder.bind(SecurityFilterChainFactory.class, SecurityFilterChainFactoryImpl.class);
080                binder.bind(ComponentRequestFilter.class, SecurityComponentRequestFilter.class);
081//              binder.bind(ShiroExceptionHandler.class);
082                binder.bind(LoginContextService.class, LoginContextServiceImpl.class);
083        }

085        public static RememberMeManager buildRememberMeManager() {
086                CookieRememberMeManager rememberMeManager = new CookieRememberMeManager();
087                // the default Shiro serializer produces obnoxiously long cookies
088                rememberMeManager.setSerializer(new SimplePrincipalSerializer());
089                return rememberMeManager;
090        }

101        /**
102         * Create ClassInterceptorsCache through annotations on the class page,
103         * which then will use SecurityFilter.
104         * <p/>
105         */
106        public void contributeApplicationInitializer(OrderedConfiguration<ApplicationInitializerFilter> configuration,
107                                                     final ComponentClassResolver componentClassResolver,
108                                                     final ClassInterceptorsCache classInterceptorsCache)
109        {
110
111                configuration.add("SecurityApplicationInitializerFilter", new ApplicationInitializerFilter()
112                {
113                        @Override
114                        public void initializeApplication(Context context, ApplicationInitializer initializer)
115                        {
116
117                                initializer.initializeApplication(context);
118
119                                for (String name : componentClassResolver.getPageNames())
120                                {
121                                        String className = componentClassResolver.resolvePageNameToClassName(name);
122                                        Class<?> clazz = ClassUtils.forName(className);
123
124                                        while (clazz != null)
125                                        {
126                                                for (Class<? extends Annotation> annotationClass : AopHelper.getAutorizationAnnotationClasses())
127                                                {
128                                                        Annotation classAnnotation = clazz.getAnnotation(annotationClass);
129                                                        if (classAnnotation != null)
130                                                        {
131                                                                //Add in the cache which then will be used in RequestFilter
132                                                                classInterceptorsCache.add(className, new DefaultSecurityInterceptor(classAnnotation));
133                                                        }
134                                                }
135                                                clazz = clazz.getSuperclass();
136                                        }
137                                }
138                        }
139                });
140        }

160        /**
161         * Secure all service methods that are marked with authorization annotations.
162         * <p/>
163         * <b>Restriction:</b> Only service interfaces can be annotated.
164         */
165        @Match("*")
166        @Order("before:*")
167        public static void adviseSecurityAssert(MethodAdviceReceiver receiver,
168                        final @Core Environment environment)
169        {
170                Class<?> serviceInterface = receiver.getInterface();
171
172                for (Method method : serviceInterface.getMethods())
173                {
174
175                        List<SecurityInterceptor> interceptors =
176                                        AopHelper.createSecurityInterceptorsSeeingInterfaces(method, serviceInterface);
177
178                        for (final SecurityInterceptor interceptor : interceptors)
179                        {
180                                MethodAdvice advice = new MethodAdvice()
181                                {
182                                        @Override
183                                        public void advise(MethodInvocation invocation)
184                                        {
185                                                // Only (try to) intercept if subject is bound.
186                                                // This is useful in case background or initializing operations
187                                                // call service operations that are secure
188                                                if (ThreadContext.getSubject() != null)
189                                                {
190                                                        environment.push(MethodInvocation.class, invocation);
191                                                        try
192                                                        {
193                                                                interceptor.intercept();
194                                                        }
195                                                        finally
196                                                        {
197                                                                environment.pop(MethodInvocation.class);
198                                                        }
199                                                }
200                                                invocation.proceed();
201
202                                        }
203                                };
204                                receiver.adviseMethod(method, advice);
205                        }
206
207                }
208        }
