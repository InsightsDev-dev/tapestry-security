001package org.tynamo.security.services.impl;
002
003import org.apache.shiro.util.AntPathMatcher;
004import org.apache.shiro.util.PatternMatcher;
005import org.apache.shiro.util.RegExPatternMatcher;
006import org.apache.tapestry5.SymbolConstants;
007import org.apache.tapestry5.ioc.annotations.EagerLoad;
008import org.apache.tapestry5.ioc.annotations.Inject;
009import org.apache.tapestry5.ioc.annotations.Symbol;
010import org.apache.tapestry5.ioc.services.PipelineBuilder;
011import org.slf4j.Logger;
012import org.tynamo.security.SecuritySymbols;
013import org.tynamo.security.internal.services.LoginContextService;
014import org.tynamo.security.services.SecurityFilterChainFactory;
015import org.tynamo.security.shiro.AccessControlFilter;
016import org.tynamo.security.shiro.authc.AnonymousFilter;
017import org.tynamo.security.shiro.authc.BasicHttpAuthenticationFilter;
018import org.tynamo.security.shiro.authc.FormAuthenticationFilter;
019import org.tynamo.security.shiro.authc.UserFilter;
020import org.tynamo.security.shiro.authz.NotFoundFilter;
021import org.tynamo.security.shiro.authz.PermissionsAuthorizationFilter;
022import org.tynamo.security.shiro.authz.PortFilter;
023import org.tynamo.security.shiro.authz.RolesAuthorizationFilter;
024import org.tynamo.security.shiro.authz.SslFilter;
025
026// Eager load since this service initializes the global filter defaults
027@EagerLoad
028public class SecurityFilterChainFactoryImpl implements SecurityFilterChainFactory {
029
030        private final Logger logger;
031
032        private final PipelineBuilder builder;
033
034        private final LoginContextService loginContextService;
035
036        private final AntPathMatcher antPathMatcher = new AntPathMatcher();
037        private final RegExPatternMatcher regExPatternMatcher= new RegExPatternMatcher();
038
039        public SecurityFilterChainFactoryImpl(Logger logger, PipelineBuilder builder, LoginContextService loginContextService,
040      @Inject @Symbol(SymbolConstants.TAPESTRY_VERSION) String tapestryVersion,
041      @Inject @Symbol(SecuritySymbols.SUCCESS_URL) String successUrl,
042      @Inject @Symbol(SecuritySymbols.LOGIN_URL) String loginUrl,
043      @Inject @Symbol(SecuritySymbols.UNAUTHORIZED_URL) String unauthorizedUrl
044        ) {
045                this.builder = builder;
046                this.logger = logger;
047                this.loginContextService = loginContextService;
048                AccessControlFilter.TAPESTRY_VERSION = tapestryVersion;
049                AccessControlFilter.LOGIN_URL = loginUrl;
050                AccessControlFilter.SUCCESS_URL = successUrl;
051                AccessControlFilter.UNAUTHORIZED_URL = unauthorizedUrl;
052        }
053
054        public SecurityFilterChain.Builder createChain(String path) {
055                return new SecurityFilterChain.Builder(logger, builder, path, antPathMatcher);
056        }
057
058        public SecurityFilterChain.Builder createChain(String pattern, PatternMatcher patternMatcher) {
059                return new SecurityFilterChain.Builder(logger, builder, pattern, patternMatcher);
060        }
061
062        public SecurityFilterChain.Builder createChainWithAntPath(String path) {
063                return new SecurityFilterChain.Builder(logger, builder, path, antPathMatcher);
064        }
065
066        public SecurityFilterChain.Builder createChainWithRegEx(String pattern) {
067                return new SecurityFilterChain.Builder(logger, builder, pattern, regExPatternMatcher);
068        }
069
070        public AnonymousFilter anon() {
071                String name = "anon";
072                AnonymousFilter filter = new AnonymousFilter(loginContextService);
073                filter.setName(name);
074                return filter;
075        }
076        
077        public NotFoundFilter notfound() {
078                return new NotFoundFilter();
079        }
080        
081
082        public UserFilter user() {
083                String name = "user";
084                UserFilter filter = new UserFilter(loginContextService);
085                filter.setName(name);
086                return filter;
087        }
088
089        public FormAuthenticationFilter authc() {
090                String name = "authc";
091                FormAuthenticationFilter filter = new FormAuthenticationFilter(loginContextService);
092                filter.setName(name);
093                return filter;
094        }
095
096        public BasicHttpAuthenticationFilter basic() {
097                String name = "authcBasic";
098                BasicHttpAuthenticationFilter filter = new BasicHttpAuthenticationFilter(loginContextService);
099                filter.setName(name);
100                return filter;
101        }
102
103        public RolesAuthorizationFilter roles() {
104                String name = "roles";
105                RolesAuthorizationFilter filter = new RolesAuthorizationFilter(loginContextService);
106                filter.setName(name);
107                return filter;
108        }
109        
110        
111        public PermissionsAuthorizationFilter perms() {
112                String name = "perms";
113                PermissionsAuthorizationFilter filter = new PermissionsAuthorizationFilter(loginContextService);
114                filter.setName(name);
115                return filter;
116        }
117
118        @Override
119        public SslFilter ssl() {
120                SslFilter filter = new SslFilter(loginContextService);
121                filter.setName("ssl");
122                return filter;
123        }
124
125        @Override
126        public PortFilter port() {
127                PortFilter filter = new PortFilter(loginContextService);
128                filter.setName("port");
129                return filter;
130        }
131
132}
