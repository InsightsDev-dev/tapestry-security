037/**
038 * Superclass for any filter that controls access to a resource and may redirect the user to the login page
039 * if they are not authenticated.  This superclass provides the method
040 * {@link #saveRequestAndRedirectToLogin(javax.servlet.ServletRequest, javax.servlet.ServletResponse)}
041 * which is used by many subclasses as the behavior when a user is unauthenticated.
042 * 
043 * This class and the subclasses that are used as Shiro's built-in filters were copied from Shiro 1.1.0
044 * and modified locally to implement same behavior as specified in https://issues.apache.org/jira/browse/SHIRO-256
045 * We'll revert to using Shiro's filters if the feature gets implemented in Shiro 2.x
046 *
047 * @since 0.4.0
048 */
049public abstract class AccessControlFilter extends AdviceFilter {
050        // default values - populated by the ChainFactory from symbols 
051        public static String TAPESTRY_VERSION, LOGIN_URL, SUCCESS_URL, UNAUTHORIZED_URL;
052        public static boolean REDIRECT_TO_SAVED_URL;
053

058                        }

071                /*
072                 * Same as {@link #setConfig()} except throws an {@link IllegalArgumentException} if config is already set.
073                 * Mostly used in the initialization logic  
074                 */
075                public void addConfig(String config) {
076                        if (config == null && this.config == null && configElements != null) return;
077                        if (config != null && config.equals(this.config)) return;
078                        if (configElements != null) throw new IllegalArgumentException("Configuration is already add for this filter, existing config is " + this.configElements + ". Use setConfig if you want to override the existing configuration");
079                        setConfig(config);
080                }
081
082                
083                public void setConfig(String config) {
084      if (config != null) configElements = split(config);
085      else configElements = new String[0];
086      this.config = config;
087                }

185    /**
186     * Returns <code>true</code> if the request is allowed to proceed through the filter normally, or <code>false</code>
187     * if the request should be handled by the
188     * {@link #onAccessDenied(ServletRequest,ServletResponse,Object) onAccessDenied(request,response,mappedValue)}
189     * method instead.
190     *
191     * @param request     the incoming <code>ServletRequest</code>
192     * @param response    the outgoing <code>ServletResponse</code>
193     * @param mappedValue the filter-specific config value mapped to this filter in the URL rules mappings.
194     * @return <code>true</code> if the request should proceed through the filter normally, <code>false</code> if the
195     *         request should be processed by this filter's
196     *         {@link #onAccessDenied(ServletRequest,ServletResponse,Object)} method instead.
197     * @throws Exception if an error occurs during processing.
198     */
199    protected abstract boolean isAccessAllowed(ServletRequest request, ServletResponse response, Object mappedValue) throws Exception;

201    /**
202     * Processes requests where the subject was denied access as determined by the
203     * {@link #isAccessAllowed(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Object) isAccessAllowed}
204     * method, retaining the {@code mappedValue} that was used during configuration.
205     * <p/>
206     * This method immediately delegates to {@link #onAccessDenied(ServletRequest,ServletResponse)} as a
207     * convenience in that most post-denial behavior does not need the mapped config again.
208     *
209     * @param request     the incoming <code>ServletRequest</code>
210     * @param response    the outgoing <code>ServletResponse</code>
211     * @param mappedValue the config specified for the filter in the matching request's filter chain.
212     * @return <code>true</code> if the request should continue to be processed; false if the subclass will
213     *         handle/render the response directly.
214     * @throws Exception if there is an error processing the request.
215     * @since 1.0
216     */
217    protected boolean onAccessDenied(ServletRequest request, ServletResponse response, Object mappedValue) throws Exception {
218        return onAccessDenied(request, response);
219    }

221    /**
222     * Processes requests where the subject was denied access as determined by the
223     * {@link #isAccessAllowed(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Object) isAccessAllowed}
224     * method.
225     *
226     * @param request  the incoming <code>ServletRequest</code>
227     * @param response the outgoing <code>ServletResponse</code>
228     * @return <code>true</code> if the request should continue to be processed; false if the subclass will
229     *         handle/render the response directly.
230     * @throws Exception if there is an error processing the request.
231     */
232    protected abstract boolean onAccessDenied(ServletRequest request, ServletResponse response) throws Exception;
233
234    /**
235     * Returns <code>true</code> if
236     * {@link #isAccessAllowed(ServletRequest,ServletResponse,Object) isAccessAllowed(Request,Response,Object)},
237     * otherwise returns the result of
238     * {@link #onAccessDenied(ServletRequest,ServletResponse,Object) onAccessDenied(Request,Response,Object)}.
239     *
240     * @return <code>true</code> if
241     *         {@link #isAccessAllowed(javax.servlet.ServletRequest, javax.servlet.ServletResponse, Object) isAccessAllowed},
242     *         otherwise returns the result of
243     *         {@link #onAccessDenied(javax.servlet.ServletRequest, javax.servlet.ServletResponse) onAccessDenied}.
244     * @throws Exception if an error occurs.
245     */
246    public boolean onPreHandle(ServletRequest request, ServletResponse response, Object mappedValue) throws Exception {
247        return isAccessAllowed(request, response, mappedValue) || onAccessDenied(request, response, mappedValue);
248    }
249    
250    protected boolean preHandle(ServletRequest request, ServletResponse response) throws Exception {
251          return onPreHandle(request, response, configElements);
252    }

254    /**
255     * Returns <code>true</code> if the incoming request is a login request, <code>false</code> otherwise.
256     * <p/>
257     * The default implementation merely returns <code>true</code> if the incoming request matches the configured
258     * {@link #getLoginUrl() loginUrl} by calling
259     * <code>{@link #pathsMatch(String, String) pathsMatch(loginUrl, request)}</code>.
260     *
261     * @param request  the incoming <code>ServletRequest</code>
262     * @param response the outgoing <code>ServletResponse</code>
263     * @return <code>true</code> if the incoming request is a login request, <code>false</code> otherwise.
264     */
265    protected boolean isLoginRequest(ServletRequest request, ServletResponse response) {
266      return pathMatcher.matches(getLoginUrl(), WebUtils.getPathWithinApplication(WebUtils.toHttp(request)));
267    }

269    /**
270     * Convenience method for subclasses to use when a login redirect is required.
271     * <p/>
272     * This implementation simply calls {@link #saveRequest(javax.servlet.ServletRequest) saveRequest(request)}
273     * and then {@link #redirectToLogin(javax.servlet.ServletRequest, javax.servlet.ServletResponse) redirectToLogin(request,response)}.
274     *
275     * @param request  the incoming <code>ServletRequest</code>
276     * @param response the outgoing <code>ServletResponse</code>
277     * @throws IOException if an error occurs.
278     */
279    protected void saveRequestAndRedirectToLogin(ServletRequest request, ServletResponse response) throws IOException {
280        saveRequest(request);
281        redirectToLogin(request, response);
282    }

286     * {@link WebUtils#saveRequest(javax.servlet.ServletRequest) WebUtils.saveRequest(request)} to save the request

292     * saveRequestAndRedirectToLogin(request,response)} directly.
293     *
294     * @param request the incoming ServletRequest to save for re-use later (for example, after a redirect).

302    /**
303     * Convenience method for subclasses that merely acquires the {@link #getLoginUrl() getLoginUrl} and redirects
304     * the request to that url.
305     * <p/>
306     * <b>N.B.</b>  If you want to issue a redirect with the intention of allowing the user to then return to their
307     * originally requested URL, don't use this method directly.  Instead you should call
308     * {@link #saveRequestAndRedirectToLogin(javax.servlet.ServletRequest, javax.servlet.ServletResponse)
309     * saveRequestAndRedirectToLogin(request,response)}, which will save the current request state so that it can
310     * be reconstructed and re-used after a successful login.
311     *
312     * @param request  the incoming <code>ServletRequest</code>
313     * @param response the outgoing <code>ServletResponse</code>
314     * @throws IOException if an error occurs.
315     */
316    protected void redirectToLogin(ServletRequest request, ServletResponse response) throws IOException {
317//        String loginUrl = getLoginUrl();
318        String localeName = loginContextService.getLocaleFromPath(WebUtils.getPathWithinApplication(WebUtils.toHttp(request)));
319        String loginUrl = localeName == null ? '/' + loginContextService.getLoginPage() : '/' + localeName + '/' + loginContextService.getLoginPage();
320        
321        // We are not in the response pipeline yet, and it's possible that Tapestry isn't handling this response, but it's still probably 
322        // better than sending a 302 and the full the page
323        if ("XMLHttpRequest".equals(WebUtils.toHttp(request).getHeader("X-Requested-With"))) {
324                WebUtils.toHttp(response).setContentType("application/json;charset=UTF-8");
325                OutputStream os = WebUtils.toHttp(response).getOutputStream();
326                    if (TAPESTRY_VERSION.startsWith("5.4")) {
327                            os.write(("{\"_tapestry\":{\"redirectURL\":\"" + WebUtils.toHttp(request).getContextPath() + loginUrl + "\"}}").getBytes());
328                    } else {
329                            os.write(("{\"redirectURL\":\"" + WebUtils.toHttp(request).getContextPath() + loginUrl + "\"}").getBytes());
330                    }
331                    os.close();
332        }
333        else WebUtils.issueRedirect(request, response, loginUrl);
334    }
