019package org.tynamo.security.shiro.authc;
020
021import javax.servlet.ServletRequest;
022import javax.servlet.ServletResponse;
023
024import org.apache.shiro.authc.AuthenticationException;
025import org.apache.shiro.authc.AuthenticationToken;
026import org.apache.shiro.authc.UsernamePasswordToken;
027import org.apache.shiro.subject.Subject;
028import org.tynamo.security.internal.services.LoginContextService;
029
030/**
031 * An <code>AuthenticationFilter</code> that is capable of automatically performing an authentication attempt
032 * based on the incoming request.
033 *
034 * @since 0.4.0
035 */
036public abstract class AuthenticatingFilter extends AuthenticationFilter {
037        public AuthenticatingFilter(LoginContextService loginContextService) {
038                super(loginContextService);
039        }
040
041    //TODO - complete JavaDoc
042
043    protected boolean executeLogin(ServletRequest request, ServletResponse response) throws Exception {
044        AuthenticationToken token = createToken(request, response);
045        if (token == null) {
046            String msg = "createToken method implementation returned null. A valid non-null AuthenticationToken " +
047                    "must be created in order to execute a login attempt.";
048            throw new IllegalStateException(msg);
049        }
050        try {
051            Subject subject = getSubject(request, response);
052            subject.login(token);
053            return onLoginSuccess(token, subject, request, response);
054        } catch (AuthenticationException e) {
055            return onLoginFailure(token, e, request, response);
056        }
057    }
058
059    protected abstract AuthenticationToken createToken(ServletRequest request, ServletResponse response) throws Exception;
060
061    protected AuthenticationToken createToken(String username, String password,
062                                              ServletRequest request, ServletResponse response) {
063        boolean rememberMe = isRememberMe(request);
064        String host = getHost(request);
065        return createToken(username, password, rememberMe, host);
066    }
067
068    protected AuthenticationToken createToken(String username, String password,
069                                              boolean rememberMe, String host) {
070        return new UsernamePasswordToken(username, password, rememberMe, host);
071    }
072
073    protected boolean onLoginSuccess(AuthenticationToken token, Subject subject,
074                                     ServletRequest request, ServletResponse response) throws Exception {
075        return true;
076    }
077
078    protected boolean onLoginFailure(AuthenticationToken token, AuthenticationException e,
079                                     ServletRequest request, ServletResponse response) {
080        return false;
081    }
082
083    /**
084     * Returns the host name or IP associated with the current subject.  This method is primarily provided for use
085     * during construction of an <code>AuthenticationToken</code>.
086     * <p/>
087     * The default implementation merely returns {@link ServletRequest#getRemoteHost()}.
088     *
089     * @param request the incoming ServletRequest
090     * @return the <code>InetAddress</code> to associate with the login attempt.
091     */
092    protected String getHost(ServletRequest request) {
093        return request.getRemoteHost();
094    }
095
096    /**
097     * Returns <code>true</code> if &quot;rememberMe&quot; should be enabled for the login attempt associated with the
098     * current <code>request</code>, <code>false</code> otherwise.
099     * <p/>
100     * This implementation always returns <code>false</code> and is provided as a template hook to subclasses that
101     * support <code>rememberMe</code> logins and wish to determine <code>rememberMe</code> in a custom mannner
102     * based on the current <code>request</code>.
103     *
104     * @param request the incoming ServletRequest
105     * @return <code>true</code> if &quot;rememberMe&quot; should be enabled for the login attempt associated with the
106     *         current <code>request</code>, <code>false</code> otherwise.
107     */
108    protected boolean isRememberMe(ServletRequest request) {
109        return false;
110    }
111}
